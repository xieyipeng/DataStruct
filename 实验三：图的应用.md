#include<stdio.h>
#include<Windows.h>
#define FINITY 5000
#define M 20
typedef char vertextype;
typedef int edgetype;
typedef struct {
	vertextype vexs[M];
	edgetype edges[M][M];
	int n, e;
}Mgraph;
void create(Mgraph m) {
	int w;//权值
	int i, j;
	printf("please input the vertices'n' and the edge 'e':\n");
	scanf("%d%d", &m.n, &m.e);
	printf("please input the vertices's value:\n");
	for (i = 0; i < m.n; i++) {//顶点
		m.vexs[i] = getchar();
	}
	for (i = 0; i < m.n; i++) {//边的初始化
		for (int j = 0; j < m.n; j++) {
			m.edges[i][j] = 0;
		}	
	}
	for (i = 0; i < m.e; i++) {//给边赋权值
		printf("please input the 'i' and 'j' and value\n");
		scanf("%d%d%d", &i, &j, &w);
		m.edges[i][j] = w;
		m.edges[j][i] = w;//无向图
	}
}
int main() {
	Mgraph mgraph;
	int n, e;
	create(mgraph);
}
