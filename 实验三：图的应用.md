#include<stdio.h>
#include<Windows.h>
#define FINITY 5000
#define M 20
#define Maxint 32767
typedef int edgetype;
typedef struct {
	char vexs[M];
	edgetype edges[M][M];
	int n, e;
}Mgraph;
int D[FINITY][FINITY], P[FINITY][FINITY];
Mgraph* create(Mgraph *m,int n,int e) {
	int i, j;
	int w;//权值
	printf("please input the vertices's value:\n");
	for (int i = 1; i <= m->n; i++) {
		//printf("please input %dth vertices\n",i);
		scanf("%s", &m->vexs[i]);
	}
	printf("this vertices is over\n");
	for (i = 1; i <= m->n; i++) {//边的初始化
		for (j = 1; j <= m->n; j++) {
			m->edges[i][j] = 0;
		}	
	}
	for (int l = 1; l <= m->e; l++) {//给边赋权值
		//printf("please input the %dth 'i' and 'j' and value\n",l);
		scanf("%d%d%d", &i, &j, &w);
		m->edges[i][j] = w;
		m->edges[j][i] = w;
	}
	printf("create over!\n");
	return m;
}
void Djkstra(Mgraph *g, int v0, int n) {
	int dist[FINITY], path[FINITY], i, j, k, min, v, x;
	boolean final[FINITY];
	for (v = 0; v < g->n; v++) {
		final[v] = FALSE;
		dist[v] = g->edges[v0][v];
		if (dist[v] < FINITY&&dist[v] != 0) {
			path[v] = v0;
		}
		else {
			path[v] = -1;
		}
	}
	final[v0] = TRUE;
	dist[v0] = 0;
	printf("first %d \n", min);
	for (i = 1; i < g->n; i++) {
		printf("hello\n");
		min = FINITY;
		printf("second %d \n", min);
		for (k = 0; k < g->n; ++k) {
			if (!final[k] && dist[k] < min) {
				v = k;
				min = dist[k];
			}
		}
	}
	printf("\n%c---%d\n", g->vexs[v], min);
	if (min == FINITY) {
		return;
	}
	else {
		final[v] = TRUE;
	}
	for (k = 0; k < g->n; ++k) {
		if (!final[k] && (min + g->edges[v][k] < dist[k])) {
			dist[k] = min + g->edges[v][k];
			path[k] = v;
		}
	}
	int st[M], pre, top = -1;
	for (i = 0; i < g->n; i++) {
		printf("\nDistanced: %d, path:", dist[i]);
		st[++top] = i;
		pre = path[i];
		while (pre != -1) {
			st[++top] = pre;
			pre = path[pre];
		}
		while (top > 0) {
			printf("%d", st[top--]);
		}
	}
}
void Floyd(Mgraph *m,int n) {
	for (int i = 1; i <= n; i++) {
		for (int j = 0; i <= n; i++) {
			if (m->edges[i][j] != Maxint) {
				m->edges[i][j] = j;
			}
			else {
				m->edges[i][j] = 0;
				D[i][j] = m->edges[i][j];
			}
		}
	}
	for (int k = 1; k <= n; k++) {
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n; j++) {
				if (D[i][k] + D[k][j] < D[i][j]) {
					D[i][j] = D[i][k] + D[k][j];
					P[i][j] = P[i][k];
					printf("D i j = %d , P i j = %d", D[i][j], P[i][j]);
				}
			}
		}
	}
	printf("please input the start and the end\n");
	int s, e;
	scanf("%d%d", &s, &e);
	if (P[s][e] == 0) {
		printf("null\n");
	}
	else {
		printf("the shortest rood is:\n");
		while (P[s][e] != e) {
			printf("->%d", P[s][e]);
			P[s][e] = P[P[s][e]][e];
		}
		printf("->%d", e);
		printf("the shortest rood is%d\n", D[s][e]);
	}
}
