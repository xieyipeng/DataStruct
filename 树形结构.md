# **孩子表示法（指针方式）**
## **前序遍历**
```java
#include<stdio.h>
#include<Windows.h>
#define m 3//度数为3
typedef char datatype;
typedef struct node {
	datatype data;
	struct node *child[m];
}node, *tree;
tree root;//root表示指向根结点的指针
node* createtree() {//必须按前序遍历的顺序输入节点的值，遇到空时，用#代替
	char ch;
	tree t;
	if ((ch = getchar()) == '#') {
		t = NULL;
	}
	else {
		t = (tree)malloc(sizeof(node));
		t->data = ch;
		for (int i = 0; i < m; i++) {
			t->child[i] = createtree();
		}
	}
	return t;
}
void preorder(tree p) {
	int i;
	if (p != NULL) {
		printf("%c", p->data);
		for (i = 0; i < m; i++) {
			preorder(p->child[i]);//递归
		}
	}
}
int main() {
	tree p;
	printf("输入各节点的值");
	printf("\n");
	p = createtree();//AB###CE###FH###I####G###D###,按前序遍历的顺序输入节点的值
	preorder(p);//输出：ABCEFHIGD
	system("pause");
	return 0;
}
```
## **层次遍历**
```java
#include<stdio.h>
#include<Windows.h>
#define m 3//度数为3
typedef char datatype;
typedef struct node {
	datatype data;
	struct node *child[m];
}node, *tree;
tree root;//root表示指向根结点的指针

void levelorder(tree t) {//层次遍历
	tree queue[100];//存放等待节点的队列
	int f = 0, r = 1;
	tree p;
	queue[0] = t;
	while (f < r) {//队列不为空
		//printf("\n");
		//printf("%d", f);
		p = queue[f];
		f++;//当f大于r时，跳出循环，此时queue[]已经遍历完成
		printf("%c", p->data);
		for (int i = 0; i < m; i++) {
			//printf("%d", i);
			if (p->child[i]) {
				queue[r] = p->child[i];
				++r;
			}
		}
	}
}
node* createtree() {//必须按前序遍历的顺序输入节点的值，遇到空时，用#代替
	char ch;
	tree t;
	if ((ch = getchar()) == '#') {
		t = NULL;
	}
	else {
		t = (tree)malloc(sizeof(node));
		t->data = ch;
		for (int i = 0; i < m; i++) {
			t->child[i] = createtree();
		}
	}
	return t;
}
int main() {
	tree p;
	printf("输入各节点的值");
	printf("\n");
	p = createtree();//AB###CE###FH###I####G###D###,按前序遍历的顺序输入节点的值
	printf("遍历结果");
	printf("\n");
	levelorder(p);//输出：ABCDEFGHI
	system("pause");
	return 0;
}
```
