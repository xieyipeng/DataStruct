# **孩子表示法（指针方式）:实现前序遍历**
```java
#include<stdio.h>
#include<Windows.h>
#define m 3//度数为3
typedef char datatype;
typedef struct node {
	datatype data;
	struct node *child[m];
}node, *tree;
tree root;//root表示指向根结点的指针
node* createtree() {//必须按前序遍历的顺序输入节点的值，遇到空时，用#代替
	char ch;
	tree t;
	if ((ch = getchar()) == '#') {
		t = NULL;
	}
	else {
		t = (tree)malloc(sizeof(node));
		t->data = ch;
		for (int i = 0; i < m; i++) {
			t->child[i] = createtree();
		}
	}
	return t;
}
void preorder(tree p) {
	int i;
	if (p != NULL) {
		printf("%c", p->data);
		for (i = 0; i < m; i++) {
			preorder(p->child[i]);//递归
		}
	}
}
int main() {
	tree p;
	printf("输入各节点的值");
	printf("\n");
	p = createtree();//AB###CE###FH###I####G###D###,按前序遍历的顺序输入节点的值
	preorder(p);//输出：ABCEFHIGD
	system("pause");
	return 0;
}
```
