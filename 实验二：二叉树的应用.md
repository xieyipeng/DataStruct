#include<stdio.h>
#include<Windows.h>
typedef int datatype;
typedef struct node {
	char brand[20];
	datatype price;
	struct node *lchild, *rchild;
}bsnode;
bsnode* insert(bsnode *root,char bra[20],int pri) {
	bsnode *t;
	t = (bsnode*)malloc(sizeof(bsnode));
	t->lchild = NULL;
	t->rchild = NULL;
	if (root == NULL) {
		printf("It is null\n");
		
		strcpy(t->brand, bra);
		t->price = pri;
		t->lchild = NULL;
		t->rchild = NULL;
		return t;
	}
	else {
		printf("It is not null\n");
		if (pri < root->price) {
			if (root->lchild == NULL) {//t的左孩子为空时，添加
				
				strcpy(t->brand, bra);
				t->price = pri;
				t->lchild = NULL;
				t->rchild = NULL;
				root->lchild = t;
				printf("insert the left\n");
			}
			else {//t的左孩子不为空时，递归
				t = insert(root->lchild, bra, pri);
			}
		}
		else {
			if (pri > root->price) {
				if (root->rchild == NULL) {//t的右孩子为空时，添加
				t = (bsnode*)malloc(sizeof(bsnode));
				strcpy(t->brand, bra);
				t->price = pri;
				t->lchild = NULL;
				t->rchild = NULL;
				root->rchild = t;
				printf("insert the right\n");
			}
			else {//t的右孩子不为空时，递归
				t = insert(root->rchild, bra, pri);
			}
			}	
		}
	}
	return root;
}
bsnode* CreatBstree(int n) {
	char bra[20];
	int pri;
	bsnode *t = NULL;
	for (int i = 0; i < n; i++) {
		printf("please input the %dth brand and price\n",i+1);
		scanf("%s%d", &bra, &pri);
		printf("hello\n");
		if (t != NULL) {
			insert(t, bra, pri);
		}
		else {
			t = insert(t, bra, pri);
		}
	}
	return t;
}
void outputTree(bsnode *t) {//前序遍历，是有序
	if (t != NULL) {
		outputTree(t->lchild);//先左
		printf("%s %d", t->brand,t->price);
		printf("\n");
		outputTree(t->rchild);//后右
		free(t);
	}
}
int main() {
	int n;
	printf("please input the number of the brand\n");
	scanf("%d", &n);
	bsnode *b1;
	b1 = CreatBstree(n);
	outputTree(b1);
	system("pause");
	return 0;
}
