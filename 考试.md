* **括号匹配**
```java
#include<stdio.h>
#include<Windows.h>
#define maxsize 100
typedef char datatype;
typedef struct {
	datatype a[maxsize];
	int top;
}stack;
int empty(stack s) {
	return (s.top ? 0 : 1);//  **0表示非空，1表示空**
}
void push(stack *s,datatype x) {
	if (s->top == maxsize) {
		printf("full!\n");
		exit(1);
	}
	s->a[s->top] = x;
	s->top++;
}
datatype read(stack s) {
	if (empty(s)) {
		printf("empty\n");
		exit(1);
	}
	return s.a[s.top - 1];// **返回top-1**
}
void pop(stack *s) {
	if (s->top == 0) {
		printf("empty\n");
		exit(1);
	}
	s->top--;
}
int match_kuohao(char c[]) {
	int i = 0;
	stack s;
	s.top = 0;
	while (c[i] != '#') {
		switch (c[i]){
		case '{':
		case '[':
		case '(':
			push(&s, c[i]);
			break;
		case ')':
			if (!empty(s) && read(s) == '(') {
				pop(&s);
				break;
			}
		case ']':
			if (!empty(s) && read(s) == '[') {
				pop(&s);
				break;
			}
		case '}':
			if (!empty(s) && read(s) == '{') {
				pop(&s);
				break;
			}
		default:
			break;
		}
		i++;
	}
	return (empty(s));// **为空则匹配**
}
int main() {
	char a[] = { '{','[','(',')',']','}','#' };
	if (match_kuohao(a) == 1) {
		printf("success!\n");
	}
	else {
		printf("fail!\n");
	}
	system("pause");
	return 0;
}
```

```java
					i++;
```

* **KMP模式匹配**
```java
#include<stdio.h>
#include<Windows.h>
#define maxsize 100
typedef char datatype;
typedef struct {
	char str[maxsize];
	int length;
}seqstring;
void getnext(seqstring p,int next[]) {
	int i, j;
	next[0] = -1;
	i = 0;
	j = -1;
	while (i < p.length) {
		if (j == -1 || p.str[i] == p.str[j]) {
			//++i;
			//++j;
			i++;
			j++;
			next[i] = j;
		}
		else {
			j = next[j];
		}
	}
	for (i = 0; i < p.length; i++) {
		printf("next数组中第%d个的值：%d\n", i + 1, next[i]);
	}
}
int KMP(seqstring t, seqstring p, int next[]) {
	int i, j;
	i = 0;
	j = 0;
	while (i < t.length&&j < p.length) {
		if (j == -1 || t.str[i] == p.str[j]) {
			i++;
			j++;
		}
		else {
			j = next[j];
		}
	}
	if (j == p.length) {
		return i - p.length;
	}
	else {
		return -1;
	}
}
int main() {
	int next[maxsize];
	seqstring *p;
	p = (seqstring*)malloc(sizeof(seqstring));
	p->length = 0;
	seqstring *t;
	t = (seqstring*)malloc(sizeof(seqstring));
	t->length = 0;
	printf("输入模式串p\n");
	for (int i = 0; i < 3; i++) {
		scanf("%s", &p->str[i]);
		p->length++;
	}
	printf("输入模正文串t\n");
	for (int i = 0; i < 9; i++) {
		scanf("%s", &t->str[i]);
		t->length++;
	}
	getnext(*p, next);
	printf("匹配结果: %d\n", KMP(*t, *p, next));
	system("pause");
	return 0;
}
```
