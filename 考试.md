* **括号匹配**
```java
#include<stdio.h>
#include<Windows.h>
#define maxsize 100
typedef char datatype;
typedef struct {
	datatype a[maxsize];
	int top;
}stack;
int empty(stack s) {
	return (s.top ? 0 : 1);//  **0表示非空，1表示空**
}
void push(stack *s,datatype x) {
	if (s->top == maxsize) {
		printf("full!\n");
		exit(1);
	}
	s->a[s->top] = x;
	s->top++;
}
datatype read(stack s) {
	if (empty(s)) {
		printf("empty\n");
		exit(1);
	}
	return s.a[s.top - 1];// **返回top-1**
}
void pop(stack *s) {
	if (s->top == 0) {
		printf("empty\n");
		exit(1);
	}
	s->top--;
}
int match_kuohao(char c[]) {
	int i = 0;
	stack s;
	s.top = 0;
	while (c[i] != '#') {
		switch (c[i]){
		case '{':
		case '[':
		case '(':
			push(&s, c[i]);
			break;
		case ')':
			if (!empty(s) && read(s) == '(') {
				pop(&s);
				break;
			}
		case ']':
			if (!empty(s) && read(s) == '[') {
				pop(&s);
				break;
			}
		case '}':
			if (!empty(s) && read(s) == '{') {
				pop(&s);
				break;
			}
		default:
			break;
		}
		i++;
	}
	return (empty(s));// **为空则匹配**
}
int main() {
	char a[] = { '{','[','(',')',']','}','#' };
	if (match_kuohao(a) == 1) {
		printf("success!\n");
	}
	else {
		printf("fail!\n");
	}
	system("pause");
	return 0;
}
```

```java
#include<stdio.h>
#include<Windows.h>
#define maxsize 100
typedef char datatype;
typedef struct {
	datatype a[maxsize];
	int top;
}stack;
double readnumber(char f[],int i) {
	double x = 0.0;
	x = f[i] - 48;
	printf("x : %f \n", x);
	return (x);
}
double evalpost(char f[]) {
	double obst[maxsize];
	int top = 0;
	int i = 0;
	double x1, x2;
	while (f[i] != '#') {
		printf("第%d个字符：%c\n", i+1,f[i]);
		if (f[i] >= '0'&&f[i] <= '9') {
			obst[top] = readnumber(f, i);
			top++;
			i++;
		}
		else {
			/*if (f[i] = ' ') {
				printf("空格\n");
				i++;
			}
			else {*/
				if (f[i] == '+') {
					printf("为+号\n");
					x2 = obst[--top];
					x1 = obst[--top];
					obst[top] = x1 + x2;
					top++;
					i++;
				}
				else {
					if (f[i] == '-') {
						printf("为-号\n");
						x2 = obst[--top];
						x1 = obst[--top];
						obst[top] = x1 - x2;
						printf("%d\n", top);
						printf("%f \n", obst[top]);
						top++;
						printf("%d\n", top);
						i++;
					}
					else {
						if (f[i] == '*') {
							printf("为*号\n");
							x2 = obst[--top];
							x1 = obst[--top];
							obst[top] = x1 * x2;
							printf("%d\n", top);
							printf("%f \n", obst[top]);
							top++;
							printf("%d\n", top);
							i++;
						}
						else {
							if (f[i] == '/') {
								printf("为/号\n");
								x2 = obst[--top];
								x1 = obst[--top];
								obst[top] = x1 / x2;
								top++;
								i++;
							}
						}
					}
				}
			}
		}
	//}
	return obst[0];
}
int main() {
	char a[] = { '3','5','2','*','-','#' };
	printf("最终结果: %f\n", evalpost(a));
	system("pause");
	return 0;
}
```
