#include<stdio.h>
#include<Windows.h>
#include <stdlib.h>
struct link_node{
	char name[20];
	char phone[20];
	struct link_node *next;
}*head;
link_node insert(){
	link_node *q;
	q = head;
	while(q->next){
		q = q->next;
	}
	link_node *p;
	p = (link_node*)malloc(sizeof(link_node));
	printf("\n");
	printf("输入要插入的同学姓名和电话：");
	printf("\n");
	scanf("%s%s", &p->name, &p->phone);
	q->next = p;
	p->next = NULL;//不要忘记将p制空
	return *head;
}
link_node dele() {
	char name[20];
	printf("输入想要删除的学生的姓名：");
	printf("\n");
	scanf("%s", name);
	link_node *p,*pre;
	p = head;
	while ((p->name) != name&&p!=NULL) {
		pre = p;
		p = p->next;
	}
	if (p) {
		if (!pre) {//头结点
			head->next = p->next;
		}
		else {//中间或者最后
			pre->next = p->next;
			//free(p);
		}
	}
	return *head;
}
void creat(int n) {
	head = new link_node;
	head->next = NULL;
	struct link_node *q, *tail;
	tail = head;
	for (int i = 0; i < n;i++) {
		q = new link_node;
		q->next = NULL;
		printf("输入第%d个学生姓名与电话", i+1);
		scanf("%s%s", &q->name, &q->phone);//注意&符号
		tail->next = q;
		tail = q;
	}
}
void Output()
{
	struct link_node *p;
	p = head->next;//必须是head->next
	printf("\n");
	while (p)
	{
		printf("%s %s", p->name, p->phone);
		p = p->next;
		printf("\n");
	}
}
int main() {
	int n;
	printf("全班有人数为：");
	scanf("%d", &n);
	creat(n);
	printf("全班通讯录为：");
	Output();
	//printf(head->name);//乱码
	//printf(head->next->name);//第一个学生
	//printf(head->next->next->name);//第二个学生
	printf("在最后的同学后面插入一个学生的通讯录：");
	insert();
	printf("插入一个学生后通讯录为：");
	Output();
	printf("输入想要删除的学生的姓名：");
	dele();
	Output();
	system("pause");
	return 0;
}
