#include<stdio.h>
#include<Windows.h>
typedef struct link_node {
	char IDnumber;
	char name[20];
	char department[20];
	int stuclass;
	struct link_node *next;
}node;
node* create(node *head,int n) {
	head = (node*)malloc(sizeof(node));
	node *p,*tail;
	tail = head;
	for (int i = 0; i < n; i++) {
		p = (node*)malloc(sizeof(node));
		printf("please input the %dth IDnumber,name,department and stuclass\n", i + 1);
		scanf("%s%s%s%d", &p->IDnumber, p->name, p->department, &p->stuclass);
		tail->next = p;
		tail = p;
	}
	return head;
}
node* display(node *head) {
	node *p;
	p = head->next;
	printf("the message of this class is:\n");
	while (p!=NULL) {
		printf("%s",p->IDnumber);
		printf("%s", p->name);
		printf("%s", p->department);
		printf("%d", p->stuclass);
		p = p->next;
		printf("\n");
	}
	return head;
}
int main() {
	int n;
	printf("how many message you want to input?\n");
	scanf("%d", &n);
	node *no;
	no = create(no, n);
	display(no);

}
